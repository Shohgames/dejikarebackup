        const startYear = 2016;
        const startMonth = 5;
        const endYear = 2025;
        const endMonth = 8;

        function generateHeaders() {
            const yearHeadersDiv = document.getElementById('yearHeaders');
            const monthHeadersDiv = document.getElementById('monthHeaders');
            const gridLinesDiv = document.getElementById('gridLines');
            
            let monthCount = 0;
            
            for (let year = startYear; year <= endYear; year++) {
                const start = (year === startYear) ? startMonth : 1;
                const end = (year === endYear) ? endMonth : 12;
                const months = end - start + 1;
                
                const yearDiv = document.createElement('div');
                yearDiv.className = 'year';
                yearDiv.textContent = year;
                yearDiv.style.flex = months;
                yearHeadersDiv.appendChild(yearDiv);
                
                for (let month = start; month <= end; month++) {
                    const monthDiv = document.createElement('div');
                    monthDiv.className = 'month';
                    const monthNames = ['J','F','M','A','M','J','J','A','S','O','N','D'];
                    monthDiv.textContent = monthNames[month - 1];
                    monthHeadersDiv.appendChild(monthDiv);
                    
                    const gridLine = document.createElement('div');
                    gridLine.className = month === 1 ? 'grid-line year-line' : 'grid-line';
                    gridLinesDiv.appendChild(gridLine);
                    
                    monthCount++;
                }
            }
        }

        function dateToMonthIndex(dateStr) {
            const [year, month] = dateStr.split('-').map(Number);
            const yearOffset = year - startYear;
            const monthOffset = month - startMonth;
            
            let index = 0;
            for (let y = startYear; y < year; y++) {
                const start = (y === startYear) ? startMonth : 1;
                const end = (y === endYear) ? endMonth : 12;
                index += end - start + 1;
            }
            
            if (year === startYear) {
                index += month - startMonth;
            } else {
                index += month - 1;
            }
            
            return index;
        }
